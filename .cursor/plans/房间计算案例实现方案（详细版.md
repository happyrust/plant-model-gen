```rust
async fn init_database_and_config() -> Result<(DbOption, DbOptionExt)> {
    info!("初始化数据库和配置...");
    
    // 1. 加载配置
    let db_option = get_db_option();
    let db_option_ext = get_db_option_ext();
    
    // 2. 验证配置
    if db_option.project_path.is_empty() {
        anyhow::bail!("项目路径未配置");

    let room_keywords = db_option.get_room_key_word();
    if room_keywords.is_empty() {
        warn!("房间关键词为空，将无法查询房间");

    info!("  项目路径: {}", db_option.project_path);
    info!("  房间关键词: {:?}", room_keywords);
    info!("  网格路径: {:?}", db_option.get_meshes_path());
    
    Ok((db_option, db_option_ext))
}
```

```rust
async fn demo_model_generation(
    db_option_ext: &DbOptionExt,
    refnos: Vec<RefnoEnum>,
) -> Result<()> {
    if refnos.is_empty() {
        warn!("参考号列表为空，跳过模型生成");
        return Ok(());
    }
    
    info!("开始生成 {} 个 refno 的模型数据", refnos.len());
    
    // 准备配置
    let mut config = db_option_ext.clone();
    config.gen_mesh = true;
    config.gen_model = true;
    config.apply_boolean_operation = Some(true);
    config.replace_mesh = Some(false);
    
    let start_time = Instant::now();
    
    // 调用模型生成
    match gen_all_geos_data(refnos.clone(), &config, None, None).await {
        Ok(success) => {
            let duration = start_time.elapsed();
            if success {
                info!("✓ 模型生成成功，耗时: {:?}", duration);
                
                // 验证生成结果
                verify_model_generation(&refnos).await?;
            } else {
                warn!("模型生成被跳过（可能已存在）");
            }
            Ok(())
        }
        Err(e) => {
            error!("✗ 模型生成失败: {}", e);
            Err(e).context("模型生成失败")
        }
    }
}

async fn verify_model_generation(refnos: &[RefnoEnum]) -> Result<()> {
    let sql = format!(
        "SELECT count() FROM inst_relate WHERE refno IN [{}]",
        refnos.iter()
            .map(|r| format!("{}", r.to_pe_key()))
            .join(", ")
    );
    
    let mut response = SUL_DB.query(&sql).await?;
    let count: Option<u64> = response.take(0)?;
    info!("  生成的几何实例数量: {:?}", count);
    
    Ok(())
}
```

```rust
async fn demo_point_query() -> Result<()> {
    info!("--- 点查询示例 ---");
    
    let test_points = vec![
        Vec3::new(1000.0, 2000.0, 3000.0),
        Vec3::new(1500.0, 2500.0, 3500.0),
        Vec3::new(2000.0, 3000.0, 4000.0),
    ];
    
    for (idx, point) in test_points.iter().enumerate() {
        info!("查询点 #{}: ({:.2}, {:.2}, {:.2})", idx + 1, point.x, point.y, point.z);
        
        let query_start = Instant::now();
        
        match query_room_number_by_point_v2(*point).await {
            Ok(Some(room_number)) => {
                info!("  ✓ 找到房间号: {}", room_number);
                
                match query_room_panel_by_point_v2(*point).await {
                    Ok(Some(panel_refno)) => {
                        info!("  ✓ 找到面板: {}", panel_refno.refno());
                    }
                    Ok(None) => warn!("  ⚠ 未找到面板"),
                    Err(e) => error!("  ✗ 查询面板失败: {}", e),
                }
            }
            Ok(None) => warn!("  ⚠ 未找到房间"),
            Err(e) => error!("  ✗ 查询失败: {}", e),
        }
        
        info!("  查询耗时: {:?}", query_start.elapsed());
    }
    
    Ok(())
}

async fn demo_batch_query() -> Result<()> {
    info!("--- 批量查询示例 ---");
    
    let test_points = vec![
        Vec3::new(1000.0, 2000.0, 3000.0),
        Vec3::new(1500.0, 2500.0, 3500.0),
        Vec3::new(2000.0, 3000.0, 4000.0),
        Vec3::new(2500.0, 3500.0, 4500.0),
        Vec3::new(3000.0, 4000.0, 5000.0),
    ];
    
    info!("批量查询 {} 个点", test_points.len());
    let batch_start = Instant::now();
    
    match batch_query_room_numbers(test_points.clone(), 8).await {
        Ok(room_numbers) => {
            let batch_time = batch_start.elapsed();
            info!("✓ 批量查询完成，耗时: {:?}", batch_time);
            
            for (i, (point, room_number)) in test_points.iter().zip(room_numbers.iter()).enumerate() {
                match room_number {
                    Some(room_num) => {
                        info!("  点 #{}: ({:.2}, {:.2}, {:.2}) -> 房间: {}", 
                              i + 1, point.x, point.y, point.z, room_num);
                    }
                    None => {
                        info!("  点 #{}: ({:.2}, {:.2}, {:.2}) -> 未找到房间", 
                              i + 1, point.x, point.y, point.z);
                    }
                }
            }
        }
        Err(e) => {
            error!("✗ 批量查询失败: {}", e);
            return Err(e).context("批量查询失败");
        }
    }
    
    Ok(())
}

    let stats = get_room_query_stats().await;
```

### 6. 辅助函数实现

**查询房间参考号函数**:

```rust
async fn query_room_refnos_for_demo(
    room_keywords: &Vec<String>,
) -> Result<Vec<(RefnoEnum, String, Vec<RefnoEnum>)>> {
    if room_keywords.is_empty() {
        return Ok(Vec::new());
    }
    
    let filter = room_keywords
        .iter()
        .map(|x| format!("'{}' in NAME", x))
        .join(" or ");
    
    #[cfg(feature = "project_hd")]
    let sql = format!(
        r#"
        select value [  id,
                        array::last(string::split(NAME, '-')),
                        array::flatten([REFNO<-pe_owner<-pe, REFNO<-pe_owner<-pe<-pe_owner<-pe])[?noun='PANE']
                    ] from FRMW where {} LIMIT 10
        "#,
        filter
    );
    
    #[cfg(not(feature = "project_hd"))]
    let sql = format!(
        r#"
        select value [  id,
                        array::last(string::split(NAME, '-')),
                        array::flatten([REFNO<-pe_owner<-pe])[?noun='PANE']
                    ] from SBFR where {} LIMIT 10
        "#,
        filter
    );
    
    let mut response = SUL_DB.query(&sql).await?;
    let raw_result: Vec<(RecordId, String, Vec<RecordId>)> = response.take(0)?;
    
    let room_groups: Vec<(RefnoEnum, String, Vec<RefnoEnum>)> = raw_result
        .into_iter()
        .map(|(room_id, room_num, panel_ids)| {
            let room_refno = RefnoEnum::from(room_id);
            let panel_refnos = panel_ids
                .into_iter()
                .map(|id| RefnoEnum::from(id))
                .collect();
            (room_refno, room_num, panel_refnos)
        })
        .collect();
    
    Ok(room_groups)
}
```

```rust
async fn main() -> Result<()> {
    // 初始化日志

    println!("=========================================");
    println!("房间计算案例演示程序");
    println!("=========================================\n");

        Ok(result) => result,
        Err(e) => {
            eprintln!("✗ 初始化失败: {}", e);
            return Err(e);
        }
    };
    println!();
    
    // 2. 查询房间参考号
    println!("步骤 2: 查询房间参考号");

    let room_keywords = db_option.get_room_key_word();
    let room_refnos = match query_room_refnos_for_demo(&room_keywords).await {

        }

    println!("找到 {} 个房间参考号\n", room_refnos.len());
    
    // 3. 模型生成示例
    println!("步骤 3: 模型生成示例");
    println!("----------------------------");
    if !room_refnos.is_empty() {
        let demo_refnos: Vec<RefnoEnum> = room_refnos.iter().take(5)

        warn!("未找到房间参考号，跳过模型生成");
    }
    println!();
    
    // 4. 关系构建示例
    println!("步骤 4: 构建房间关系");

        warn!("sqlite 特性未启用，跳过关系构建");
```

### 9. 错误处理策略

1. **分层错误处理**: 每个模块独立处理错误，使用 `anyhow::Context` 添加上下文
2. **优雅降级**: 特性未启用时给出警告而非错误
3. **详细日志**: 使用 `tracing` 记录每个步骤的详细信息
