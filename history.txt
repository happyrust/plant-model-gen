-- 测试新的查询方式：通过 OWNER 字段查询
-- 从 FRMW 查询 -> SBFR (OWNER = FRMW.REFNO) -> PANE (OWNER = SBFR.REFNO)
USE NS `1516` DB AvevaMarineSample;
-- 简化查询：从 FRMW 到 PANE 的层级
-- Step 1: 找到 FRMW
LET $frmw = (select id, NAME, REFNO from FRMW where NAME IS NOT NONE AND '-RM' in NAME limit 1)[0];
RETURN $frmw;
-- Step 2: 找到 SBFR (OWNER = FRMW.REFNO)
LET $sbfrs = (select id, REFNO from SBFR where OWNER = $frmw.REFNO);
RETURN $sbfrs;
-- Step 3: 找到 PANE (OWNER = SBFR.REFNO)
LET $panes = (select id, REFNO from PANE where OWNER IN array::map($sbfrs, |$s: any| $s.REFNO));
RETURN $panes;
