#### 2. use-tasks Hook

**职责**: 获取任务列表数据，处理加载状态

**接口设计**:
```typescript
interface UseTasksOptions {
  status?: TaskStatus | "all"
  limit?: number
  autoRefresh?: boolean
  refreshInterval?: number  // 毫秒
}

interface UseTasksReturn {
  tasks: Task[]
  total: number
  loading: boolean
  error: string | null
  stats: TaskStats
  refetch: () => Promise<void>
}

interface TaskStats {
  total: number
  pending: number
  running: number
  completed: number
  failed: number
  cancelled: number
}
```

**实现示例**:
```typescript
// hooks/use-tasks.ts
import { useState, useEffect, useCallback, useMemo } from "react"
import { fetchTasks } from "@/lib/api/tasks"
import type { Task, TaskStatus } from "@/types/task"

export function useTasks(options: UseTasksOptions = {}): UseTasksReturn {
  const { status = "all", limit, autoRefresh = true, refreshInterval = 5000 } = options
  
  const [tasks, setTasks] = useState<Task[]>([])
  const [total, setTotal] = useState(0)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const fetchData = useCallback(async () => {
    setLoading(true)
    setError(null)

    try {
      const data = await fetchTasks({
        status: status === "all" ? undefined : status,
        limit,
      })
      
      setTasks(data.tasks)
      setTotal(data.total)
    } catch (err) {
      setError(err instanceof Error ? err.message : "获取任务列表失败")
    } finally {
      setLoading(false)
    }
  }, [status, limit])

  // 初始加载
  useEffect(() => {
    fetchData()
  }, [fetchData])

  // 自动刷新
  useEffect(() => {
    if (!autoRefresh) return

    const interval = setInterval(fetchData, refreshInterval)
    return () => clearInterval(interval)
  }, [autoRefresh, refreshInterval, fetchData])

  // 计算统计信息
  const stats = useMemo(() => ({
    total,
    pending: tasks.filter(t => t.status === "Pending").length,
    running: tasks.filter(t => t.status === "Running").length,
    completed: tasks.filter(t => t.status === "Completed").length,
    failed: tasks.filter(t => t.status === "Failed").length,
    cancelled: tasks.filter(t => t.status === "Cancelled").length,
  }), [tasks, total])

  return {
    tasks,
    total,
    loading,
    error,
    stats,
    refetch: fetchData,
  }
}
```

### 🔑 技术实现要点

1. **轮询优化**
   - 仅在有Running状态任务时启用轮询
   - 页面不可见时暂停轮询（使用 `document.visibilityState`）
   - 使用 `useRef` 存储定时器ID，避免重复创建

2. **性能优化**
   - 使用 `useMemo` 缓存计算结果
   - 使用 `useCallback` 避免函数重复创建
   - 任务列表使用虚拟滚动（如果数量>100）

3. **错误处理**
   - API调用失败时显示Toast提示
   - 网络错误时自动重试（最多3次）
   - 显示友好的错误信息

4. **用户体验**
   - 乐观更新（操作立即反映在UI）
   - 操作失败时回滚状态
   - 加载状态使用骨架屏

### 🔧 组件设计

#### TaskLogsViewer 组件

**Props接口**:
```typescript
interface TaskLogsViewerProps {
  taskId: string
  autoRefresh?: boolean
  refreshInterval?: number
  maxLogs?: number
}
```

**数据结构**:
```typescript
interface LogEntry {
  timestamp: number  // Unix timestamp (ms)
  level: LogLevel
  message: string
  error_code: string | null
  stack_trace: string | null
}

type LogLevel = "Debug" | "Info" | "Warning" | "Error" | "Critical"
```

**实现示例**:
```typescript
// components/tasks/task-logs-viewer.tsx
"use client"

import { useState, useEffect, useRef } from "react"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { Input } from "@/components/ui/input"
import { Button } from "@/components/ui/button"
import { Select } from "@/components/ui/select"
import { Download, Pause, Play, X } from "lucide-react"
import { useTaskLogs } from "@/hooks/use-task-logs"
import { LogEntry } from "./log-entry"
import type { LogLevel } from "@/types/task"

export function TaskLogsViewer({ taskId, autoRefresh = true, refreshInterval = 10000, maxLogs = 200 }: TaskLogsViewerProps) {
  const [levelFilter, setLevelFilter] = useState<LogLevel | "all">("all")
  const [searchQuery, setSearchQuery] = useState("")
  const [isPaused, setIsPaused] = useState(false)
  const logsEndRef = useRef<HTMLDivElement>(null)

  const { logs, task, loading, refetch } = useTaskLogs(taskId, {
    autoRefresh: autoRefresh && !isPaused,
    refreshInterval,
    limit: maxLogs,
  })

  // 筛选日志
  const filteredLogs = logs.filter(log => {
    if (levelFilter !== "all" && log.level !== levelFilter) return false
    if (searchQuery && !log.message.toLowerCase().includes(searchQuery.toLowerCase())) return false
    return true
  })

  // 自动滚动到底部
  useEffect(() => {
    if (!isPaused) {
      logsEndRef.current?.scrollIntoView({ behavior: "smooth" })
    }
  }, [logs, isPaused])

  const handleExport = () => {
    const content = filteredLogs.map(log => 
      `[${new Date(log.timestamp).toISOString()}] [${log.level}] ${log.message}`
    ).join("\n")
    
    const blob = new Blob([content], { type: "text/plain" })
    const url = URL.createObjectURL(blob)
    const a = document.createElement("a")
    a.href = url
    a.download = `task-${taskId}-logs.txt`
    a.click()
    URL.revokeObjectURL(url)
  }

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>任务日志</CardTitle>
          <div className="flex gap-2">
            <Button size="sm" variant="outline" onClick={() => setIsPaused(!isPaused)}>
              {isPaused ? <Play className="h-4 w-4" /> : <Pause className="h-4 w-4" />}
            </Button>
            <Button size="sm" variant="outline" onClick={handleExport}>
              <Download className="h-4 w-4 mr-1" />
              导出
            </Button>
          </div>
        </div>
      </CardHeader>
      <CardContent>
        {/* 筛选器 */}
        <div className="flex gap-2 mb-4">
          <Select value={levelFilter} onValueChange={setLevelFilter}>
            <option value="all">全部级别</option>
            <option value="Debug">Debug</option>
            <option value="Info">Info</option>
            <option value="Warning">Warning</option>
            <option value="Error">Error</option>
            <option value="Critical">Critical</option>
          </Select>
          <Input
            placeholder="搜索日志..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
          />
          {(levelFilter !== "all" || searchQuery) && (
            <Button size="sm" variant="ghost" onClick={() => {
              setLevelFilter("all")
              setSearchQuery("")
            }}>
              <X className="h-4 w-4" />
            </Button>
          )}
        </div>

        {/* 日志列表 */}
        <div className="bg-slate-950 rounded-lg p-4 h-96 overflow-y-auto font-mono text-sm">
          {filteredLogs.length === 0 ? (
            <div className="text-center text-slate-500 py-8">暂无日志</div>
          ) : (
            filteredLogs.map((log, index) => (
              <LogEntry key={index} log={log} />
            ))
          )}
          <div ref={logsEndRef} />
        </div>

        <div className="mt-2 text-sm text-muted-foreground">
          显示 {filteredLogs.length} / {logs.length} 条日志
        </div>
      </CardContent>
    </Card>
  )
}
```

#### DbNumberInput 组件

**实现示例**:
```typescript
// components/tasks/db-number-input.tsx
"use client"

import { useState } from "react"
import { Input } from "@/components/ui/input"
import { Button } from "@/components/ui/button"
import { Badge } from "@/components/ui/badge"
import { X } from "lucide-react"

interface DbNumberInputProps {
  value: number[]
  onChange: (value: number[]) => void
}

export function DbNumberInput({ value, onChange }: DbNumberInputProps) {
  const [inputValue, setInputValue] = useState("")

  const parseInput = (input: string): number[] => {
    const nums: number[] = []
    const parts = input.split(/[,，\s]+/).filter(Boolean)

    for (const part of parts) {
      // 支持范围格式：7999-8005
      if (part.includes("-")) {
        const [start, end] = part.split("-").map(Number)
        if (!isNaN(start) && !isNaN(end) && start <= end) {
          for (let i = start; i <= end; i++) {
            nums.push(i)
          }
        }
      } else {
        const num = Number(part)
        if (!isNaN(num)) {
          nums.push(num)
        }
      }
    }

    return [...new Set(nums)].sort((a, b) => a - b)
  }

  const handleAdd = () => {
    const newNums = parseInput(inputValue)
    if (newNums.length > 0) {
      onChange([...new Set([...value, ...newNums])].sort((a, b) => a - b))
      setInputValue("")
    }
  }

  const handleRemove = (num: number) => {
    onChange(value.filter(n => n !== num))
  }

  return (
    <div className="space-y-2">
      <div className="flex gap-2">
        <Input
          placeholder="输入数据库号，支持逗号分隔或范围（如：7999-8005）"
          value={inputValue}
          onChange={(e) => setInputValue(e.target.value)}
          onKeyDown={(e) => {
            if (e.key === "Enter") {
              e.preventDefault()
              handleAdd()
            }
          }}
        />
        <Button onClick={handleAdd}>添加</Button>
      </div>

      {value.length > 0 && (
        <div className="flex flex-wrap gap-2">
          {value.map(num => (
            <Badge key={num} variant="secondary">
              {num}
              <button
                className="ml-1 hover:text-destructive"
                onClick={() => handleRemove(num)}
              >
                <X className="h-3 w-3" />
              </button>
            </Badge>
          ))}
        </div>
      )}
    </div>
  )
}
```

```typescript
// hooks/use-task-history.ts
import { useState, useEffect, useCallback } from "react"
import { fetchTasks } from "@/lib/api/tasks"
import type { Task } from "@/types/task"

interface UseTaskHistoryOptions {
  status?: "completed" | "failed" | "cancelled" | "all"
  taskType?: string
  dateFrom?: string
  dateTo?: string
  limit?: number
}

export function useTaskHistory(options: UseTaskHistoryOptions = {}) {
  const [tasks, setTasks] = useState<Task[]>([])
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const fetchHistory = useCallback(async () => {
    setLoading(true)
    setError(null)

    try {
      const data = await fetchTasks({
        status: options.status === "all" ? undefined : options.status,
        limit: options.limit,
      })

      // 前端筛选（如果后端不支持）
      let filteredTasks = data.tasks

      if (options.taskType) {
        filteredTasks = filteredTasks.filter(t => t.task_type === options.taskType)
      }

      if (options.dateFrom) {
        const fromDate = new Date(options.dateFrom).getTime()
        filteredTasks = filteredTasks.filter(t => t.created_at >= fromDate)
      }

      if (options.dateTo) {
        const toDate = new Date(options.dateTo).getTime()
        filteredTasks = filteredTasks.filter(t => t.created_at <= toDate)
      }

      setTasks(filteredTasks)
    } catch (err) {
      setError(err instanceof Error ? err.message : "获取历史记录失败")
    } finally {
      setLoading(false)
    }
  }, [options])

  useEffect(() => {
    fetchHistory()
  }, [fetchHistory])

  return {
    tasks,
    loading,
    error,
    refetch: fetchHistory,
  }
}
```

**实现示例**:
```typescript
// hooks/use-task-stats.ts
import { useMemo } from "react"
import type { Task } from "@/types/task"

export function useTaskStats(tasks: Task[]) {
  return useMemo(() => {
    const total = tasks.length
    const completed = tasks.filter(t => t.status === "Completed").length
    const failed = tasks.filter(t => t.status === "Failed").length
    const cancelled = tasks.filter(t => t.status === "Cancelled").length

    const success_rate = total > 0 ? (completed / total) * 100 : 0

    // 计算平均执行时间
    const completedTasks = tasks.filter(t => t.actual_duration !== null)
    const total_duration = completedTasks.reduce((sum, t) => sum + (t.actual_duration || 0), 0)
    const avg_duration = completedTasks.length > 0
      ? total_duration / completedTasks.length / 1000  // 转换为秒
      : 0

    return {
      total,
      completed,
      failed,
      cancelled,
      success_rate,
      avg_duration,
      total_duration: total_duration / 1000,  // 转换为秒
    }
  }, [tasks])
}
```
